#创建分支
git branch 分支名
#切换分支
git checkout 分支名
#创建于切换同时进行
git checkout -b 分支名
#删除分支
git branch -b 分支名

#查看当前分支,当前分支会有一个*号
git branch

#合分支
git checkout master #切换要合的分支
git merge xym #将xym分支合并

########### 实例操作123
 #创建分支
git branch dev
#切换分支
git checkout dev
 #拉代码
git pull origin master
 #创建远程分支，提交代码
git push --set-upstream origin dev

#切换主分支
git checkout master
#将dev合并到master
git merge dev
#将主分支提交上去
git push --set-upstream origin master
